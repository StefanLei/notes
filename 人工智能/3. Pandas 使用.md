#### Pandas 使用

Series 是一个类数组的数据结构，同时带有标签或者说索引，索引是在前面的序号。

### Series 类型

> Series 的创建

```python
# 创建 Series
ser1 = Series([1,2,3,4])
print(ser1)

# 创建指定索引的 Series 
ser2 = Series(range(4),index=['a','b','c','d'])

# 通过字典创建 Series
sdata = {'a':1,'b':2}
ser3 = Series(sdata)
```

> Series 的元素操作

```python
ser1=Series([1,2,3,4]) 
ser2=Series(range(4),index=["a","b","c","d"]) ser3=Series({'Ohio':35000,'Texas':71000,'Oregon':16000,'Utah':5000})

# 获取指定索引的元素 
print(ser2["a"])
print(ser2[['a']])

# 同时获取多个值
print(ser2[['a','b']])


```

> Series 的运算

```python
ser = Series([1,2,3])
ser2 = ser*2  # [2,4,6]

ser3 = ser+ser2

```

### DataFrame 数据类型

> DataFrame 创建

```python
from pandas import Series, DataFrame
import pandas
import numpy

data = {
    'state': ['cn', 'usa', 'uk'],
    'year': [100, 111, 111],
    'pop': [1.5, 1.7, 1.6]
}

# key 是字段名，后面的值。序号默认是数字
df1 = DataFrame(data)

data = [
    [1, 2, 3],
    [3, 4, 5]
]

# 自定义字段名，自定义索引名
df = DataFrame(data, columns=['y', 's', 'p'], index=['one', 'two'])

s1 = numpy.array([1, 2, 3, 4])
s2 = numpy.array([5, 6, 7, 8])

# 把行向量，转换成 dataFrame
df2 = DataFrame([s1, s2])

# 把行向量转换成 series
Series(numpy.array([1, 2, 3, 4]))

s1 = Series([1, 2, 3])
s2 = Series([4, 5, 6])

# 把series转换成 DataFrame
DataFrame([s1, s2])

# 类型不同的 DataFrame
df = pandas.DataFrame(
    {'A': 1.,
     'B': pandas.Timestamp('20130102'),
     'C': pandas.Series(1, index=list(range(4)), dtype='float32'),
     'D': numpy.array([2] * 4, dtype='int32'),
     'E': pandas.Categorical(["test", "train", "test", "train"]),
     'F': 'foo'}
)

# 从文件中读取 创建 DataFrame, 第一行的数据会当作字段
df = pandas.read_csv("E7.csv")
print(df)
```

> 常用属性

```python
from pandas import Series, DataFrame
import pandas
import numpy

ser = Series([1, 2, 3, 4], index=['a', 'b', 'c', 'd'])

df = DataFrame({
    'a': [1, 2, 3, 4],
    'b': [2, 3, 4, 5]
})

# 返回所有的字段
print(df.columns)

# 返回 DataFrame 的形状 (4,2) 4行2列
print(df.shape)

# 返回所有的索引 index
print(df.index)

# 返回一个二维的数组，没有 index 和 columns
print(df.values)

```
