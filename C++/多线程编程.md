#### C++ 多线程编程

##### 互斥锁

```cpp
#include "iostream"
#include "thread"
#include "vector"
#include "mutex"
#include "chrono"
#include "stdexcept"

using namespace std;

int counter = 0;

// 互斥锁
mutex mtx;

void increase(int time) {
    for (int i = 0; i < time; i++) {
        // 当前线程休眠 1 毫秒
        this_thread::sleep_for(chrono::microseconds(1));
        mtx.lock();
        counter++;
        mtx.unlock();
    }
}

int main() {
    thread t1(increase, 10000);
    thread t2(increase, 10000);

    t1.join();
    t2.join();

    cout << "counter: " << counter << endl;
}
```

##### 直接使用 mtx 容易造成死锁

```cpp
#include <iostream>
#include <thread>
#include <vector>
#include <mutex>
#include <chrono>
#include <stdexcept>

int counter = 0;
std::mutex mtx; // 保护counter

void increase_proxy(int time, int id) {
    for (int i = 0; i < time; i++) {
        mtx.lock();
        // 线程1上锁成功后，抛出异常：未释放锁
        if (id == 1) {
            throw std::runtime_error("throw excption....");
        }
        // 当前线程休眠1毫秒
        std::this_thread::sleep_for(std::chrono::milliseconds(1));
        counter++;
        mtx.unlock();
    }
}

void increase(int time, int id) {
    try {
        increase_proxy(time, id);
    }
    catch (const std::exception &e) {
        std::cout << "id:" << id << ", " << e.what() << std::endl;
    }
}

int main() {
    std::thread t1(increase, 10000, 1);
    std::thread t2(increase, 10000, 2);
    t1.join();
    t2.join();
    std::cout << "counter:" << counter << std::endl;
    return 0;
}
```

##### 配合 lock_guard 使用

```cpp
#include <iostream>
#include <thread>
#include <vector>
#include <mutex>
#include <chrono>
#include <stdexcept>

int counter = 0;
std::mutex mtx; // 保护counter

void increase_proxy(int time, int id) {
    for (int i = 0; i < time; i++) {
        // std::lock_guard对象构造时，自动调用mtx.lock()进行上锁
        // std::lock_guard对象析构时，自动调用mtx.unlock()释放锁
        std::lock_guard<std::mutex> lk(mtx);
        // 线程1上锁成功后，抛出异常：未释放锁
        if (id == 1) {
            throw std::runtime_error("throw excption....");
        }
        // 当前线程休眠1毫秒
        std::this_thread::sleep_for(std::chrono::milliseconds(1));
        counter++;
    }
}

void increase(int time, int id) {
    try {
        increase_proxy(time, id);
    }
    catch (const std::exception& e){
        std::cout << "id:" << id << ", " << e.what() << std::endl;
    }
}

int main(int argc, char** argv) {
    std::thread t1(increase, 10000, 1);
    std::thread t2(increase, 10000, 2);
    t1.join();
    t2.join();
    std::cout << "counter:" << counter << std::endl;
    return 0;
}
```

##### 条件变量 案例一

```cpp
#include <iostream>
#include <mutex>
#include <thread>
#include <vector>
#include <condition_variable>

using namespace std;


mutex myMutex;
condition_variable cv;
string data;
bool ready = false;
bool processed = false;


void Worker() {
    unique_lock<mutex> lock(myMutex);

    // 等待主线程发送数据
    cv.wait(lock, [] {
        cout << "worker waiting ..." << endl;
        return ready;
    });
    cout << "工作线程正在处理数据..." << endl;

    // 睡眠1秒模拟处理数据
    this_thread::sleep_for(chrono::seconds(2));
    data = data + "已处理";

    // 把数据发回主线程
    processed = true;
    cout << "工作线程通知数据已经处理完毕" << endl;

    // 通知前,手动解锁，防止正在等待的线程被唤醒后又立刻被阻塞
    lock.unlock();

    cv.notify_one();
}


int main() {
    thread worker(Worker);
    {
        lock_guard<mutex> lock(myMutex);
        cout << "主线程正在准备数据" << endl;

        // 睡眠1秒，模拟准备数据.
        this_thread::sleep_for(chrono::seconds(3));
        data = "样本数据";
        ready = true;
        cout << "主线程通知数据已经准备完毕" << endl;
    }
    cv.notify_one();

    {
        unique_lock<mutex> lock(myMutex);
        cv.wait(lock, [] {
            cout << "main waiting ..." << endl;
            return processed;
        });
    }

    cout << "回到主线程，数据 = " << data << endl;
    worker.join();
    return 0;
}
```

##### 条件变量 案例二

```cpp
#include <iostream>
#include <mutex>
#include <thread>
#include <queue>

using namespace std;

mutex myMutex;
condition_variable cond;
bool done = false;
queue<int> myQueue;

void produce() {
    for (int i = 0; i <= 100; i++) {
        unique_lock<mutex> lock(myMutex);
        cout << "生产数据 " << i << endl;
        myQueue.push(i);

        lock.unlock();
        cond.notify_one();
        this_thread::sleep_for(chrono::seconds(1));
    }
    done = true;
}


void consumer() {
    while (!done) {
        unique_lock<mutex> lock(myMutex);
        while (myQueue.empty()) {
            cond.wait(lock);
        }
        int data = myQueue.front();
        myQueue.pop();
        cout << "消费数据 " << data << endl;
        lock.unlock();
    }
}

int main() {
    thread t1(produce);
    thread t2(consumer);

    t1.join();
    t2.join();
}
```
